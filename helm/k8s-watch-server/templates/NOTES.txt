1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "k8s-watch-server.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "k8s-watch-server.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "k8s-watch-server.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "k8s-watch-server.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.config.serverPort }}
{{- end }}

2. Check the status:
  kubectl --namespace {{ .Release.Namespace }} get pods -l "app.kubernetes.io/name={{ include "k8s-watch-server.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. View logs:
  kubectl --namespace {{ .Release.Namespace }} logs -f -l "app.kubernetes.io/name={{ include "k8s-watch-server.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

4. Test the API:
  # Port forward first
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "k8s-watch-server.fullname" . }} 8080:{{ .Values.service.port }}
  
  # In another terminal:
  curl http://localhost:8080/health
  curl "http://localhost:8080/api/v1/events?limit=10"

5. Configure the MCP server:
  export AUDIT_API_URL="http://{{ include "k8s-watch-server.fullname" . }}.{{ .Release.Namespace }}.svc:{{ .Values.service.port }}"
